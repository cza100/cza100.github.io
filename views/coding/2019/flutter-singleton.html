            # 前言

            由于在 flutter 中使用了 BLoC 模式，BLoC 模式和 Navigator 依赖于Context，但是我需要在Context外的工具函数或者初始化时实现全局数据共享以及路由导航。像 flutter 的 Dio 和 SharedPreferences 也特别适合用单例模式封装。

            # Dart 中的单例模式

            在 dart 中由于有工厂构造函数，所以很容易实现单例模式，以下是 3 种简单实现的方式：

            # 1. Factory constructor

            class Singleton {

  static final Singleton _instance = Singleton._internal();

  Singleton._internal();

  factory Singleton() {
    return _instance;
  }
}

            # 简单封装 Dio

            class DioUtil {
  static final DioUtil _instance = DioUtil._internal();
  Dio _dio;

  factory DioUtil() => _instance;

  DioUtil._internal() {
    if (_dio == null) {
      BaseOptions options = new BaseOptions(
        baseUrl: "http://testApi.com.cn",
        connectTimeout: 5000,
        receiveTimeout: 3000,
      );
      _dio = new Dio(options);
    }
  }

  // 初始化时动态设置baseUrl
  void setBaseUrl(String baseUrl) {
    _dio.options.baseUrl = baseUrl;
  }

  // 在请求头中设置用户token
  void setToken(String token) {
    _dio.options.headers['token'] = token;
  }

  // GET
  get(String path, [Map<String, dynamic> params]) async {
    try {
      Response response;
      if (params != null) {
        response = await _dio.get(path, queryParameters: params);
      } else {
        response = await _dio.get(path);
      }
      return response.data;
    } on DioError catch (e) {
      if (e.response != null) {
        // The request was made and the server responded with a status code
        // that falls out of the range of 2xx and is also not 304.
      } else {
        // Something happened in setting up or sending the request that triggered an Error
      }
    }
  }

  //POST
  post(String path, Map<String, dynamic> params) async {
    try {
      Response response;
      response = await _dio.post(path, data: params);
      return response.data;
    } on DioError catch (e) {
      if (e.response != null) {
        // The request was made and the server responded with a status code
        // that falls out of the range of 2xx and is also not 304.
      } else {
        // Something happened in setting up or sending the request that triggered an Error
      }
    }
  }
}

            # 实现全局变量控制器 ConfigController

            class ConfigController {
  static final ConfigController _instance = ConfigController._internal();
  // 这里可以定义一个config类型，偷懒了
  Map<String, dynamic> _config;

  factory ConfigController() => _instance;

  ConfigController._internal() {
    if (_config == null) {
      // config初始化
      _config = {};
    }
  }

  // config setter
  void setNavigator(navigator) {
    _config['navigator'] = navigator;
  }

  Future<void> getHost() async {
    var prefs = await PrefsController.create(<String, dynamic>{
      'action': 'get',
      'key': 'host',
    });
    _config['host'] = prefs.value;
  }

  ...
  ...
  ...

  // config getter
  Map<String, dynamic> get config => _config;

            # 使用场景

            # 在 context 外使用 Navigator

            final GlobalKey<NavigatorState> navigator = new GlobalKey<NavigatorState>();
ConfigController().setNavigator(navigator);

// 在任何地方使用Navigator
// 若Dio请求返回401，跳转至登录页面
post(String path, Map<String, dynamic> params) async {
  try {
    Response response;
    response = await _dio.post(path, data: params);
    return response.data;
  } on DioError catch (e) {
    if (e.response != null) {
      if (e.response.statusCode == 401) {
        ConfigController()
          .config['navigator']
          .currentState
          .pushReplacementNamed('/login');
      }
    }
  }
}

            # Widget 构建前做一些初始化配置

            void main() async {
  // Do some initialization
  // Set Host
  await ConfigController().getHost();
  DioUtil().setBaseUrl(ConfigController().config['host']);

  // Set Token
  await ConfigController().getToken();
  DioUtil().setToken(ConfigController().config['token']);

  // Set Ap
  await ConfigController().getAp();

  // Config navigator
  final GlobalKey<NavigatorState> navigator = new GlobalKey<NavigatorState>();
  ConfigController().setNavigator(navigator);

  // Determine entry
  Widget entry = new LoginPage();
  if (ConfigController().config['token'] != null) {
    entry = new MyHomePage();
  }

  runApp(BlocProvider<AppBloc>(
    bloc: AppBloc(),
    child: new MyApp(
      entry: entry,
    ),
  ));
}
